## ğŸ“ Counting Sort by Digit (Helper)

```cpp
void countingSortByDigit(vector<int>& arr, int exp) {
    int n = arr.size();
    vector<int> output(n), count(10, 0);

    // 1) Count occurrences of each digit (0-9)
    for (int x : arr) {
        int digit = (x / exp) % 10;
        count[digit]++;
    }
    // 2) Accumulate counts (prefix sums)
    for (int i = 1; i < 10; ++i) {
        count[i] += count[i - 1];
    }
    // 3) Build output array in **reverse** for stability
    for (int i = n - 1; i >= 0; --i) {
        int digit = (arr[i] / exp) % 10;
        output[--count[digit]] = arr[i];
    }
    // 4) Copy back to arr
    for (int i = 0; i < n; ++i)
        arr[i] = output[i];
}
```

void countingSortByDigit(vector<int>& arr, int exp) {
    int n = arr.size();
    vector<int> output(n), count(10, 0);

    for (int x : arr) {
        int digit = (x / exp) % 10;
        count[digit]++;

    for (int i = 1; i < 10; ++i)
        count[i] += count[i - 1];

    for (int i = n - 1; i >= 0; --i) {

// Main Radix Sort (LSD, base 10)
void radixSort(vector<int>& arr) {
    if (arr.empty()) return;
    // 1) Find maximum to know digit count
    int maxVal = *max_element(arr.begin(), arr.end());
    // 2) Do counting sort for each digit
    for (int exp = 1; maxVal / exp > 0; exp *= 10) {
        countingSortByDigit(arr, exp);
    }
}
