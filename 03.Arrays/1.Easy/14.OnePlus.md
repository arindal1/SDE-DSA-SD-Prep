vector<int> plusOne(vector<int>& digits) {
    int n = digits.size();
    int carry = 1;  // initial +1

    // Add carry from least significant digit backwards
    for (int i = n - 1; i >= 0 && carry; --i) {
        int sum = digits[i] + carry;
        digits[i] = sum % 10;
