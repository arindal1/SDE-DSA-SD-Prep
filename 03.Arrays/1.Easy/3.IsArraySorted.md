// Returns true if arr[0..n-1] is sorted non-decreasingly
bool isArraySorted(const vector<int>& arr, int n) {
    // Edge case: empty or single-element arrays are trivially sorted
    if (n <= 1) return true;

    for (int i = 1; i < n; i++) {
        if (arr[i] < arr[i-1]) {
            return false;
        }
    }
    return true;
}
